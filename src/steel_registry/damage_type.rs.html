4/// Represents a damage type definition from a data pack JSON file.
5#[derive(Debug)]
6pub struct DamageType {
7    pub key: Identifier,
8    pub message_id: &'static str,
9    pub scaling: DamageScaling,
10    pub exhaustion: f32,
11    pub effects: DamageEffects,
12    pub death_message_type: DeathMessageType,
13}
14
15/// How the damage scales with difficulty.
16#[derive(Debug, Clone, Copy, PartialEq, Eq)]
17pub enum DamageScaling {
18    Always,
19    WhenCausedByLivingNonPlayer,
20    Never,
21}
22
23/// The sound effects played when an entity is damaged.
24#[derive(Debug, Clone, Copy, PartialEq, Eq)]
25pub enum DamageEffects {
26    Hurt,
27    Thorns,
28    Drowning,
29    Burning,
30    Poking,
31    Freezing,
32}
33
34/// How the death message is formatted.
35#[derive(Debug, Clone, Copy, PartialEq, Eq)]
36pub enum DeathMessageType {
37    Default,
38    FallVariants,
39    IntentionalGameDesign,
40}

42pub type DamageTypeRef = &'static DamageType;
43
44pub struct DamageTypeRegistry {
45    damage_types_by_id: Vec<DamageTypeRef>,
46    damage_types_by_key: FxHashMap<Identifier, usize>,
47    tags: FxHashMap<Identifier, Vec<Identifier>>,
48    allows_registering: bool,
49}
50
51impl DamageTypeRegistry {
52    #[must_use]
53    pub fn new() -> Self {
54        Self {
55            damage_types_by_id: Vec::new(),
56            damage_types_by_key: FxHashMap::default(),
57            allows_registering: true,
58            tags: FxHashMap::default(),
59        }
60    }
61
62    pub fn register(&mut self, damage_type: DamageTypeRef) -> usize {
63        assert!(
64            self.allows_registering,
65            "Cannot register damage types after the registry has been frozen"
66        );
67
68        let id = self.damage_types_by_id.len();
69        self.damage_types_by_key.insert(damage_type.key.clone(), id);
70        self.damage_types_by_id.push(damage_type);
71        id
72    }
73
74    /// Replaces damage at a given index.
75    /// Returns true if the damage was replaced and false if the damage wasn't replaced
76    #[must_use]
77    pub fn replace(&mut self, damage: DamageTypeRef, id: usize) -> bool {
78        if id >= self.damage_types_by_id.len() {
79            return false;
80        }
81        self.damage_types_by_id[id] = damage;
82        true
83    }
84
85    pub fn iter(&self) -> impl Iterator<Item = (usize, DamageTypeRef)> + '_ {
86        self.damage_types_by_id
87            .iter()
88            .enumerate()
89            .map(|(id, &dt)| (id, dt))
90    }
91}
92
93impl Default for DamageTypeRegistry {
94    fn default() -> Self {
95        Self::new()
96    }
97}
