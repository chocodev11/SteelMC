17/// Result of an interaction (item use, block use, etc.)
18#[derive(Debug, Clone, Copy, PartialEq, Eq)]
19pub enum InteractionResult {
20    /// The interaction succeeded and consumed the action.
21    Success,
22    /// The interaction failed and consumed the action.
23    Fail,
24    /// The interaction did not apply; try the next handler.
25    Pass,
26    /// Try the empty-hand interaction on the block.
27    TryEmptyHandInteraction,
28}
29
30impl InteractionResult {
31    /// Returns true if this result consumes the action (Success or Fail).
32    /// Pass and `TryEmptyHandInteraction` do not consume the action.
33    #[must_use]
34    pub const fn consumes_action(&self) -> bool {
35        matches!(self, InteractionResult::Success | InteractionResult::Fail)
36    }
37}

39/// Context for placing a block.
40pub struct BlockPlaceContext<'a> {
41    /// The position that was clicked.
42    pub clicked_pos: BlockPos,
43    /// The face of the block that was clicked.
44    pub clicked_face: Direction,
45    /// The exact location where the click occurred.
46    pub click_location: Vector3<f64>,
47    /// Whether the click was inside the block.
48    pub inside: bool,
49    /// The position where the block will be placed.
50    pub relative_pos: BlockPos,
51    /// Whether the clicked block is being replaced.
52    pub replace_clicked: bool,
53    /// The horizontal direction the player is facing.
54    pub horizontal_direction: Direction,
55    /// The player's rotation (yaw).
56    pub rotation: f32,
57    /// The player's pitch (vertical look angle).
58    pub pitch: f32,
59    /// The world where the block is being placed.
60    pub world: &'a Arc<World>,
61}
62
63impl BlockPlaceContext<'_> {
64    /// Returns the direction the player is looking at most directly.
65    ///
66    /// This considers both yaw and pitch to determine the nearest direction
67    /// among all 6 directions (UP, DOWN, NORTH, SOUTH, EAST, WEST).
68    ///
69    /// Based on Java's `Direction.orderedByNearest(Entity)[0]`.
70    #[must_use]
71    pub fn get_nearest_looking_direction(&self) -> Direction {
72        self.get_nearest_looking_directions()[0]
73    }
74
75    /// Returns all 6 directions ordered by how closely the player is looking at them.
76    ///
77    /// Based on Java's `BlockPlaceContext.getNearestLookingDirections()`.
78    /// When not replacing the clicked block, the opposite of the clicked face
79    /// is moved to the front of the array.
80    #[must_use]
81    pub fn get_nearest_looking_directions(&self) -> [Direction; 6] {
82        let mut directions = Direction::ordered_by_nearest(self.rotation, self.pitch);
83
84        // If not replacing the clicked block, prioritize the opposite of clicked face
85        if !self.replace_clicked {
86            let clicked_opposite = self.clicked_face.opposite();
87            let mut index = 0;
88
89            // Find the index of the opposite direction
90            while index < directions.len() && directions[index] != clicked_opposite {
91                index += 1;
92            }
93
94            // Move it to the front by shifting elements
95            if index > 0 && index < directions.len() {
96                // Shift elements [0..index] to [1..index+1] and put opposite at [0]
97                directions.copy_within(0..index, 1);
98                directions[0] = clicked_opposite;
99            }
100        }
101
102        directions
103    }
104
105    /// Returns true if the block at the relative position is a water source
106    #[must_use]
107    pub fn is_water_source(&self) -> bool {
108        use crate::fluid::get_fluid_state;
109        let fluid_state = get_fluid_state(self.world, self.relative_pos);
110        fluid_state.is_source() && fluid_state.is_water()
111    }
112}

114/// Context for using an item on a block.
115pub struct UseOnContext<'a> {
116    /// The player using the item.
117    pub player: &'a Player,
118    /// Which hand the item is in.
119    pub hand: InteractionHand,
120    /// Information about where the block was hit.
121    pub hit_result: BlockHitResult,
122    /// The world where the interaction is happening.
123    pub world: &'a Arc<World>,
124    /// The item stack being used (mutable for consumption).
125    pub item_stack: &'a mut ItemStack,
126    /// Lock guard holding the player's inventory.
127    pub inv_guard: &'a mut ContainerLockGuard,
128}
129
130/// Context for using an item (general usage).
131pub struct UseItemContext<'a> {
132    /// The player using the item.
133    pub player: &'a Player,
134    /// Which hand the item is in.
135    pub hand: InteractionHand,
136    /// The world where the interaction is happening.
137    pub world: &'a Arc<World>,
138    /// The item stack being used (mutable for consumption).
139    pub item_stack: &'a mut ItemStack,
140    /// Lock guard holding the player's inventory. Behaviors that need to add
141    /// items (e.g. bucket swap) must use this instead of `player.add_item_or_drop`
142    /// to avoid deadlocking on the inventory mutex.
143    pub inv_guard: &'a mut ContainerLockGuard,
144}
